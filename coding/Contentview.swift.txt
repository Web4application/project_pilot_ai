import SwiftUI

struct ContentView: View {
    @State private var data = "Loading..."

    var body: some View {
        VStack(spacing: 20) {
            Text("ðŸš€ My AI App")
                .font(.largeTitle)
                .bold()
            
            Text(data)
                .padding()
            
            Button("Refresh") {
                fetchData()
            }
        }
        .onAppear {
            fetchData()
        }
    }

    func fetchData() {
        guard let url = URL(string: "https://api.kubu-hai.com/data") else { return }
        
        URLSession.shared.dataTask(with: url) { data, _, error in
            if let data = data, let text = String(data: data, encoding: .utf8) {
                DispatchQueue.main.async {
                    self.data = text
                }
            } else if let error = error {
                DispatchQueue.main.async {
                    self.data = "Error: \(error.localizedDescription)"
                }
            }
        }.resume()
    }
}
